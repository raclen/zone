因朋友需要，写的一个滚动，挺简单的，就几行代码
```javascript
  function fun_scroll() {
        scroll = setInterval(function () {
            var $ul = $('#data_list ul');
            var $li = $ul.find("li");
            $ul.animate({
                marginTop: "-122px"
            }, 500, function () {
                $li.last().after($li.eq(0).clone());
                $li.eq(0).remove();
                $ul.css({marginTop: "0px"})
            });
        }, 2000)
    }
    fun_scroll()
    $('#data_list').mouseenter(function () {
        clearInterval(scroll);
    }).mouseleave(function () {
        fun_scroll()
    })
```
