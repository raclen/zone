## 方法1
```javascript
var arr="123456";
var str=[];
for(i=0;i<arr.length;i++){
	str.push(arr[i]);

	}
str.reverse();
var dd=str.join("");
alert(dd);
```
## 方法2
```javascript
var arr="123456";
var str="";
for(i=arr.length-1;i>=0;i--){
	str+=arr[i];
	}
alert(str);
```
