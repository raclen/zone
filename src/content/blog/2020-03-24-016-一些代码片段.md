```javascript
// 重点：replace的参数，第一个是匹配的字符，第二个是捕获的字符
var y="2020";
var m='3';
var d='24';
function render(str=''){
 return str.replace(/\${(\w+)}/g,function(arg1,arg2){
  return window[arg2]
})
}
render('${y}-${m}-${d}');
```
