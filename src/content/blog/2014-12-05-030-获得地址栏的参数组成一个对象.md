```javascript
var url = 'http://weibo.com/t2275025/home?topnav=1&wvr=5';
   var sUrl = url.substr(url.indexOf("?") + 1, url.length);
   var lct = {};
   var ourl = sUrl.split("&");
   for (var i = 0; i <ourl.length; i++) {
       var o = ourl[i].split("=");
       lct[o[0]] = o[1];

   }
   console.log(lct);
```
