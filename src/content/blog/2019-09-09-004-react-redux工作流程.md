## 下面看一下，每一步如何实现
1.action和reduces的关系
`action.js`
```javascript
export function addTodo(text) {
  return {
    type: "add.todo",
    text
  };
}
```
`reduces.js`
```javascript
function todos(state = [], action) {
  i++;
  console.log("action", action);
  if (action.type === "add.todo") {
    return [...state, { text: action.text, index: i }];
  } else if (action.type === "toggle.todo") {
    return state;
  }
  return state;
}
```
通过action.type来控制如何操作state，reduces里面就是具体的实现
2.reduces和store的关系
`reduces.js` 把所有的reduces合并到一起，准备放到store里面
```javascript
const todoApp = combineReducers({
  todos: todos,
  visibilityFilter: visibilityFilter
});
export default todoApp;
```
外层的组件 `index.js`
```javascript
import React from "react";
import ReactDOM from "react-dom";
import { Provider } from "react-redux";
import { createStore } from "redux";
import todoApp from "./reducers";
import App from "./App";
import "./styles.css";

let store = createStore(todoApp);

function Root() {
  return (
    <Provider store={store}>
      <div className="App">
        <App />
      </div>
    </Provider>
  );
}

const rootElement = document.getElementById("root");
ReactDOM.render(<Root />, rootElement);
```
注意`let store = createStore(todoApp);` 和 `Provider`
3.某容器组件
```javascript
import React, { Component } from "react";
import { connect } from "react-redux";
import { addTodo } from "../actions";

class AddTodo extends Component {
  handleAdd = () => {
    let value = this.input.value;
    this.props.dispatch(addTodo(value));
    this.input.value = "";
  };
  render() {
    return (
      <div>
        <h2>AddTodo</h2>
        <>
          <input ref={e => (this.input = e)} />
          <button onClick={this.handleAdd}>添加</button>
        </>
      </div>
    );
  }
}
export default connect(({ todos }) => ({ todos }))(AddTodo);

```
connect 后会得到 `todos` 的数据和`dispatch`方法
